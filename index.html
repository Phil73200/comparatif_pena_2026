function renderChart(data){ const labels = data.map(d=>d.Nationcode); let values = []; let datasets = []; /* ===================================================== MODE NORMAL ===================================================== */ if(currentMode !== "both"){ if(currentGender==="men"){ values = data.map(d=>d.Nb23_M); } else if(currentGender==="women"){ values = data.map(d=>d.Nb23_W); } else{ values = data.map(d=>d.Nb23_Total); } datasets = [ { label:"Courses", data:values, backgroundColor: labels.map(nation=> nation==="FRA" ? "#0055ff" : "rgba(180,180,180,0.45)" ), borderColor: labels.map(nation=> nation==="FRA" ? "#003399" : "rgba(120,120,120,0.25)" ), borderWidth:1 } ]; } /* ===================================================== MODE COMPARAISON DIRECTE ===================================================== */ else{ let valuesWithout = []; let valuesWith = []; if(currentGender==="men"){ valuesWithout = dataWithout.map(d=>d.Nb23_M); valuesWith = dataWith.map(d=>{ const found = dataWith.find( x=>x.Nationcode===d.Nationcode ); return found ? found.Nb23_M : 0; }); } else if(currentGender==="women"){ valuesWithout = dataWithout.map(d=>d.Nb23_W); valuesWith = dataWith.map(d=>{ const found = dataWith.find( x=>x.Nationcode===d.Nationcode ); return found ? found.Nb23_W : 0; }); } else{ valuesWithout = dataWithout.map(d=>d.Nb23_Total); valuesWith = dataWith.map(d=>{ const found = dataWith.find( x=>x.Nationcode===d.Nationcode ); return found ? found.Nb23_Total : 0; }); } datasets = [ { label:"Sans additif", data:valuesWithout, backgroundColor: labels.map(nation=> nation==="FRA" ? "rgba(0,85,255,0.85)" : "rgba(180,180,180,0.35)" ) }, { label:"Avec additif +8", data:valuesWith, backgroundColor: labels.map(nation=> nation==="FRA" ? "rgba(255,80,80,0.95)" : "rgba(120,120,120,0.25)" ) } ]; } if(chart){ chart.destroy(); } chart = new Chart( document.getElementById("chart"), { type:"bar", data:{ labels:labels, datasets:datasets }, options:{ responsive:true, animation:{ duration:1200 }, plugins:{ legend:{ display:true } }, scales:{ y:{ beginAtZero:true, max:GLOBAL_MAX } } } } ); } document .getElementById("btnBoth") .addEventListener("click",()=>{ currentMode = "both"; document .getElementById("btnWithout") .classList.remove("active"); document .getElementById("btnWith") .classList.remove("active"); document .getElementById("btnBoth") .classList.add("active"); update(); }); /* ===== bouton sans additif ===== */ document .getElementById("btnWithout") .addEventListener("click",()=>{ currentMode = "without"; document .getElementById("btnWithout") .classList.add("active"); document .getElementById("btnWith") .classList.remove("active"); document .getElementById("btnBoth") .classList.remove("active"); update(); }); /* ===== bouton avec additif ===== */ document .getElementById("btnWith") .addEventListener("click",()=>{ currentMode = "with"; document .getElementById("btnWith") .classList.add("active"); document .getElementById("btnWithout") .classList.remove("active"); document .getElementById("btnBoth") .classList.remove("active"); update(); });
